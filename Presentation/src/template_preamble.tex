\providecommand{\labelitemi}{$-$}

\newcommand{\litem}[3][]{%
	\item #2\\
	\vspace{-10pt}
	\begin{tikzpicture}
		% Parse the optional argument into color and length
		\def\tempcolor{HSHLDarkBlue} % Default color
		\def\templength{\textwidth/2}         % Default length
		
		\ifx\relax#1\relax % Check if the optional argument is provided
		\else
		\def\tempcolor{#1} % If yes, set the color
		\def\templength{#2} % Length is the second argument
		\fi
		
		% Draw the line with specified length and color
		\draw[\tempcolor, line width=0.5pt] (0,0) -- (\templength,0);
		
		% Draw the circle at the end of the line
		\draw[\tempcolor, fill] (\templength,0) circle (1pt);
	\end{tikzpicture}
	
			\ifx&#3&%
		% #1 is empty
		\else
		\raggedleft
		\begin{itemize}
			#3 % The content of the sublist
		\end{itemize}
		\raggedright
		
		\fi

}
